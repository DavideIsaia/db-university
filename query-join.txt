1. Selezionare tutti gli studenti iscritti al Corso di Laurea in Economia
SELECT `students`.`name` AS `nome`,`students`.`surname` AS `cognome`, `degrees`.`name` AS `corso_di_laurea`
FROM `students`
INNER JOIN `degrees`
ON `degree_id` = `degrees`.`id`
HAVING `corso_di_laurea` = 'Corso di Laurea in Economia'  
ORDER BY `cognome` ASC;

2. Selezionare tutti i Corsi di Laurea Magistrale del Dipartimento di Neuroscienze
SELECT `departments`.`name` AS `nome_dipartimento`, `degrees`.`name` AS `corso_di_laurea`, `degrees`.`level`
FROM `degrees`
INNER JOIN `departments`
ON `department_id` = `departments`.`id`
HAVING `nome_dipartimento` = 'Dipartimento di Neuroscienze' AND `degrees`.`level` = 'magistrale';
